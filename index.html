// دالة حساب النطاق بعد التعديل لضمان دقة الأرقام function calculateRange(surah, startAyah, faces) { const surahAyat = quranData.filter(x => x.sura_name_ar === surah); const startIndex = surahAyat.findIndex(x => Number(x.aya_no) === Number(startAyah)); if (startIndex === -1) return null; const startRecord = surahAyat[startIndex]; // تحويل القيم إلى أرقام لضمان نجاح العملية الحسابية const targetPage = Number(startRecord.page) + (Number(faces) / 2); let endRecord = startRecord; for (let i = startIndex; i < surahAyat.length; i++) { if (Number(surahAyat[i].page) <= targetPage) { endRecord = surahAyat[i]; } else { break; } } return { start: startRecord.aya_no, end: endRecord.aya_no, nextAyah: Number(endRecord.aya_no) + 1, finished: Number(endRecord.aya_no) === Number(surahAyat[surahAyat.length - 1].aya_no) }; } // دالة إنشاء الخطة (تأكد من استبدال دالتك القديمة بهذه لضمان التوافق) function handleGeneratePlan() { // التحقق من تحميل البيانات قبل البدء if (typeof quranData === 'undefined' || quranData.length === 0) { console.error("بيانات القرآن غير محملة"); return; } // هنا يتم جلب المدخلات من التصميم الخاص بك const selectedSurah = document.getElementById('surahSelect').value; const startAyah = document.getElementById('startAyahInput').value; const faces = document.getElementById('facesInput').value; const result = calculateRange(selectedSurah, startAyah, faces); if (result) { // تحديث واجهة المستخدم دون تغيير التنسيق document.getElementById('resultDisplay').innerText = `من آية ${result.start} إلى آية ${result.end}`; } else { alert("خطأ في تحديد الآيات، يرجى التأكد من البيانات"); } }
